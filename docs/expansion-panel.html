ExpansionPanel

ExpansionPanel

// instance
const expansionPanel = new zxEditor.ExpansionPanel({
  head: null,
  // className
  className: '',
  // head setup
  headHeight: 44,
  headLeftBtnClassName: '',
  headLeftBtnText: '',
  headTitle: 'Expansion Panel',
  // height
  height: 260,
  // body内容
  body: null,
  // Used to distinguish ExpansionPanel instance
  name: 'expansion-panel',
  onHeadClick: function () {},
  onBodyClick: function () {}
}, zxEditor)

Options

className

类型: string

默认: ''

扩展面板className。

head

类型: HTMLElement | ZxQuery | Html Code

默认: null

扩展面板head。

// HTMLElement
const head = document.createElement('div')
head.className = 'custom-expansion-panel-head-wrapper'
head.innerHTML = '<h2>扩展面板标题</h2><button>关闭按钮</button>'

head.addEventListener('click', function () {
  alert('我点击了head，下面可以做我想做的事情')
})

const expansionPanel = new zxEditor.ExpansionPanel({
  head: head
}, zxEditor)

// ZxQuery
const head = zxEditor.$('<div class="custom-expansion-panel-head-wrapper"><h2>扩展面板标题</h2><button>关闭按钮</button></div>')

const expansionPanel = new zxEditor.ExpansionPanel({
  head: head
}, zxEditor)

// Html Code
const head = '<div class="custom-expansion-panel-head-wrapper"><h2>扩展面板标题</h2><button>关闭按钮</button></div>'

const expansionPanel = new zxEditor.ExpansionPanel({
  head: head
}, zxEditor)

[!WARNING|style:flat|label:注意]

参数 headHeight 为 0 时，head不显示。

headHeight

类型: number

默认: 44

head height。

headLeftBtnClassName

类型: string

默认: ''

head 左边按钮className。

headLeftBtnText

类型: string

默认: ''

head 左边按钮className。

headTitle,

类型: string

默认: Expansion Panel

head 标题。

height

类型: string

默认: 260

扩展面板高度。

body

类型: HTMLElement | ZxQuery | Html Code

默认: null

扩展面板body部分。需要自定义内容，否则无内容显示。

name

类型: string

默认: expansion-panel

实例化后的name值。

onHeadClick

类型: function

参数: event, type

head 被点击时的回调函数。

const expansionPanel = new zxEditor.ExpansionPanel({
  onHeadClick: function (e, type) {
    console.log(e, type)
  }
}, zxEditor)

onBodyClick

类型: string

参数: event

body 被点击时的回调函数。

onBeforeShow

类型: function

参数: undefined

onShow

类型: function

参数: undefined

onBeforeHide

类型: function

参数: undefined

onHide

类型: function

参数: undefined
